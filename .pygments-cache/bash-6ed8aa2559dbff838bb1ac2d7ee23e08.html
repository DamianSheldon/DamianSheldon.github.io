// 创建标签
$ git tag <name>
$ git tag -a <tagname> -m "blablabla..."

// 查看标签
$ git tag
$ git show <tagname>

// 推送标签
$ git push origin <tagname>
$ git push origin develop:develop --tags --verbose

// 删除标签
$ git tag -d <tagname>

// 删除一个远程标签
$ git tag -d <tagname>
$ git push origin :refs/tags/<tagname>
